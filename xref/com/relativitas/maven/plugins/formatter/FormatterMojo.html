68  /**
69   * A Maven plugin mojo to format Java source code using the Eclipse code
70   * formatter.
71   * 
72   * Mojo parameters allow customizing formatting by specifying the config XML
73   * file, line endings, compiler version, and source code locations. Reformatting
74   * source files is avoided using an md5 hash of the content, comparing to the
75   * original hash to the hash after formatting and a cached hash.
76   * 
77   * @author jecki
78   * @author Matt Blanchette
79   */

248 	/**
249 	 * Execute.
250 	 *
251 	 * @throws MojoExecutionException the mojo execution exception
252 	 * @see org.apache.maven.plugin.AbstractMojo#execute()
253 	 */
254 	@Override
255 	public void execute() throws MojoExecutionException {
256 		if (this.skipFormatting) {
257 			getLog().info("Formatting is skipped");
258 			return;
259 		}
260 		long startClock = System.currentTimeMillis();
261 
262 		if (StringUtils.isEmpty(this.encoding)) {
263 			this.encoding = ReaderFactory.FILE_ENCODING;
264 			getLog().warn(
265 					"File encoding has not been set, using platform encoding ("
266 							+ this.encoding
267 							+ ") to format source files, i.e. build is platform dependent!");
268 		} else {
269 			try {
270 				"Test Encoding".getBytes(this.encoding);
271 			} catch (UnsupportedEncodingException e) {
272 				throw new MojoExecutionException("Encoding '" + this.encoding
273 						+ "' is not supported");
274 			}
275 			getLog().info(
276 					"Using '" + this.encoding + "' encoding to format source files.");
277 		}
278 
279 		if (!LINE_ENDING_AUTO.equals(this.lineEnding)
280 				&& !LINE_ENDING_KEEP.equals(this.lineEnding)
281 				&& !LINE_ENDING_LF.equals(this.lineEnding)
282 				&& !LINE_ENDING_CRLF.equals(this.lineEnding)
283 				&& !LINE_ENDING_CR.equals(this.lineEnding)) {
284 			throw new MojoExecutionException(
285 					"Unknown value for lineEnding parameter");
286 		}
287 
288 		createResourceCollection();
289 
290 		List<File> files = new ArrayList<File>();
291 		try {
292 			if (this.directories != null) {
293 				for (File directory : this.directories) {
294 					if (directory.exists() && directory.isDirectory()) {
295 						this.collection.setBaseDir(directory);
296 						addCollectionFiles(files);
297 					}
298 				}
299 			} else { // Using defaults of source main and test dirs
300 				if (this.sourceDirectory != null && this.sourceDirectory.exists()
301 						&& this.sourceDirectory.isDirectory()) {
302 					this.collection.setBaseDir(this.sourceDirectory);
303 					addCollectionFiles(files);
304 				}
305 				if (this.testSourceDirectory != null && this.testSourceDirectory.exists()
306 						&& this.testSourceDirectory.isDirectory()) {
307 					this.collection.setBaseDir(this.testSourceDirectory);
308 					addCollectionFiles(files);
309 				}
310 			}
311 		} catch (IOException e) {
312 			throw new MojoExecutionException(
313 					"Unable to find files using includes/excludes", e);
314 		}
315 
316 		int numberOfFiles = files.size();
317 		Log log = getLog();
318 		log.info("Number of files to be formatted: " + numberOfFiles);
319 
320 		if (numberOfFiles > 0) {
321 			createCodeFormatter();
322 			ResultCollector rc = new ResultCollector();
323 			Properties hashCache = readFileHashCacheFile();
324 
325 			String basedirPath = getBasedirPath();
326 			for (int i = 0, n = files.size(); i < n; i++) {
327 				File file = files.get(i);
328 				formatFile(file, rc, hashCache, basedirPath);
329 			}
330 
331 			storeFileHashCache(hashCache);
332 
333 			long endClock = System.currentTimeMillis();
334 
335 			log.info("Successfully formatted: " + rc.successCount + " file(s)");
336 			log.info("Fail to format        : " + rc.failCount + " file(s)");
337 			log.info("Skipped               : " + rc.skippedCount + " file(s)");
338 			log.info("Approximate time taken: "
339 					+ ((endClock - startClock) / 1000) + "s");
340 		}
341 	}

344 	 * Create a {@link PlexusIoFileResourceCollection} instance to be used by
345 	 * this mojo. This collection uses the includes and excludes to find the
346 	 * source files.
347 	 */
348 	void createResourceCollection() {
349 		this.collection = new PlexusIoFileResourceCollection();
350 		if (this.includes != null && this.includes.length > 0) {
351 			this.collection.setIncludes(this.includes);
352 		} else {
353 			this.collection.setIncludes(DEFAULT_INCLUDES);
354 		}
355 		this.collection.setExcludes(this.excludes);
356 		this.collection.setIncludingEmptyDirectories(false);
357 
358 		IncludeExcludeFileSelector fileSelector = new IncludeExcludeFileSelector();
359 		fileSelector.setIncludes(DEFAULT_INCLUDES);
360 		this.collection.setFileSelectors(new FileSelector[] { fileSelector });
361 	}

392 	/**
393 	 * Store file hash cache.
394 	 *
395 	 * @param props the props
396 	 */
397 	private void storeFileHashCache(Properties props) {
398 		File cacheFile = new File(this.targetDirectory, CACHE_PROPERTIES_FILENAME);
399 		try {
400 			OutputStream out = new BufferedOutputStream(new FileOutputStream(
401 					cacheFile));
402 			props.store(out, null);
403 		} catch (FileNotFoundException e) {
404 			getLog().warn("Cannot store file hash cache properties file", e);
405 		} catch (IOException e) {
406 			getLog().warn("Cannot store file hash cache properties file", e);
407 		}
408 	}

410 	/**
411 	 * Read file hash cache file.
412 	 *
413 	 * @return the properties
414 	 */
415 	private Properties readFileHashCacheFile() {
416 		Properties props = new Properties();
417 		Log log = getLog();
418 		if (!this.targetDirectory.exists()) {
419 			this.targetDirectory.mkdirs();
420 		} else if (!this.targetDirectory.isDirectory()) {
421 			log.warn("Something strange here as the "
422 					+ "supposedly target directory is not a directory.");
423 			return props;
424 		}
425 
426 		File cacheFile = new File(this.targetDirectory, CACHE_PROPERTIES_FILENAME);
427 		if (!cacheFile.exists()) {
428 			return props;
429 		}
430 
431 		try {
432 			props.load(new BufferedInputStream(new FileInputStream(cacheFile)));
433 		} catch (FileNotFoundException e) {
434 			log.warn("Cannot load file hash cache properties file", e);
435 		} catch (IOException e) {
436 			log.warn("Cannot load file hash cache properties file", e);
437 		}
438 		return props;
439 	}

442 	 * Format file.
443 	 *
444 	 * @param file the file
445 	 * @param rc the rc
446 	 * @param hashCache the hash cache
447 	 * @param basedirPath the basedir path
448 	 */
449 	private void formatFile(File file, ResultCollector rc,
450 			Properties hashCache, String basedirPath) {
451 		try {
452 			doFormatFile(file, rc, hashCache, basedirPath);
453 		} catch (IOException e) {
454 			rc.failCount++;
455 			getLog().warn(e);
456 		} catch (MalformedTreeException e) {
457 			rc.failCount++;
458 			getLog().warn(e);
459 		} catch (BadLocationException e) {
460 			rc.failCount++;
461 			getLog().warn(e);
462 		}
463 	}

465 	/**
466 	 * Format individual file.
467 	 *
468 	 * @param file the file
469 	 * @param rc the rc
470 	 * @param hashCache the hash cache
471 	 * @param basedirPath the basedir path
472 	 * @throws IOException Signals that an I/O exception has occurred.
473 	 * @throws BadLocationException the bad location exception
474 	 */
475 	private void doFormatFile(File file, ResultCollector rc,
476 			Properties hashCache, String basedirPath) throws IOException,
477 			BadLocationException {
478 		Log log = getLog();
479 		log.debug("Processing file: " + file);
480 		String code = readFileAsString(file);
481 		String originalHash = md5hash(code);
482 
483 		String canonicalPath = file.getCanonicalPath();
484 		String path = canonicalPath.substring(basedirPath.length());
485 		String cachedHash = hashCache.getProperty(path);
486 		if (cachedHash != null && cachedHash.equals(originalHash)) {
487 			rc.skippedCount++;
488 			log.debug("File is already formatted.");
489 			return;
490 		}
491 
492 		String lineSeparator = getLineEnding(code);
493 
494 		TextEdit te = this.formatter.format(CodeFormatter.K_COMPILATION_UNIT
495 				+ CodeFormatter.F_INCLUDE_COMMENTS, code, 0, code.length(), 0,
496 				lineSeparator);
497 		if (te == null) {
498 			rc.skippedCount++;
499 			log.debug("Code cannot be formatted. Possible cause "
500 					+ "is unmatched source/target/compliance version.");
501 			return;
502 		}
503 
504 		IDocument doc = new Document(code);
505 		te.apply(doc);
506 		String formattedCode = doc.get();
507 		String formattedHash = md5hash(formattedCode);
508 		hashCache.setProperty(path, formattedHash);
509 
510 		if (originalHash.equals(formattedHash)) {
511 			rc.skippedCount++;
512 			log.debug("Equal hash code. Not writing result to file.");
513 			return;
514 		}
515 
516 		writeStringToFile(formattedCode, file);
517 		rc.successCount++;
518 	}

520 	/**
521 	 * Md5hash.
522 	 *
523 	 * @param str the str
524 	 * @return the string
525 	 * @throws UnsupportedEncodingException the unsupported encoding exception
526 	 */
527 	private String md5hash(String str) throws UnsupportedEncodingException {
528 		return DigestUtils.md5Hex(str.getBytes(this.encoding));
529 	}

531 	/**
532 	 * Read the given file and return the content as a string.
533 	 *
534 	 * @param file the file
535 	 * @return the string
536 	 * @throws IOException Signals that an I/O exception has occurred.
537 	 */
538 	private String readFileAsString(File file) throws java.io.IOException {
539 		StringBuilder fileData = new StringBuilder(1000);
540 		BufferedReader reader = null;
541 		try {
542 			reader = new BufferedReader(ReaderFactory.newReader(file, this.encoding));
543 			char[] buf = new char[1024];
544 			int numRead = 0;
545 			while ((numRead = reader.read(buf)) != -1) {
546 				String readData = String.valueOf(buf, 0, numRead);
547 				fileData.append(readData);
548 				buf = new char[1024];
549 			}
550 		} finally {
551 			IOUtil.close(reader);
552 		}
553 		return fileData.toString();
554 	}

556 	/**
557 	 * Write the given string to a file.
558 	 *
559 	 * @param str the str
560 	 * @param file the file
561 	 * @throws IOException Signals that an I/O exception has occurred.
562 	 */
563 	private void writeStringToFile(String str, File file) throws IOException {
564 		if (!file.exists() && file.isDirectory()) {
565 			return;
566 		}
567 
568 		BufferedWriter bw = null;
569 		try {
570 			bw = new BufferedWriter(WriterFactory.newWriter(file, this.encoding));
571 			bw.write(str);
572 		} finally {
573 			IOUtil.close(bw);
574 		}
575 	}

587 	/**
588 	 * Return the options to be passed when creating {@link CodeFormatter}
589 	 * instance.
590 	 *
591 	 * @return the formatting options
592 	 * @throws MojoExecutionException the mojo execution exception
593 	 */
594 	private Map<String, String> getFormattingOptions()
595 			throws MojoExecutionException {
596 		Map<String, String> options = new HashMap<String, String>();
597 		options.put(JavaCore.COMPILER_SOURCE, this.compilerSource);
598 		options.put(JavaCore.COMPILER_COMPLIANCE, this.compilerCompliance);
599 		options.put(JavaCore.COMPILER_CODEGEN_TARGET_PLATFORM,
600 				this.compilerTargetPlatform);
601 
602 		if (this.configFile != null) {
603 			Map<String, String> config = getOptionsFromConfigFile();
604 			if (Boolean.TRUE.equals(this.overrideConfigCompilerVersion)) {
605 				config.remove(JavaCore.COMPILER_SOURCE);
606 				config.remove(JavaCore.COMPILER_COMPLIANCE);
607 				config.remove(JavaCore.COMPILER_CODEGEN_TARGET_PLATFORM);
608 			}
609 			options.putAll(config);
610 		}
611 
612 		return options;
613 	}

615 	/**
616 	 * Read config file and return the config as {@link Map}.
617 	 *
618 	 * @return the options from config file
619 	 * @throws MojoExecutionException the mojo execution exception
620 	 */
621 	private Map<String, String> getOptionsFromConfigFile()
622 			throws MojoExecutionException {
623 
624 		InputStream configInput = null;
625 		try {
626 			this.resourceManager.addSearchPath(FileResourceLoader.ID,
627 					this.basedir.getAbsolutePath());
628 			configInput = this.resourceManager.getResourceAsInputStream(this.configFile);
629 		} catch (ResourceNotFoundException e) {
630 			throw new MojoExecutionException("Config file [" + this.configFile
631 					+ "] cannot be found", e);
632 		}
633 
634 		if (configInput == null) {
635 			throw new MojoExecutionException("Config file [" + this.configFile
636 					+ "] does not exist");
637 		}
638 		try {
639 			ConfigReader configReader = new ConfigReader();
640 			return configReader.read(configInput);
641 		} catch (IOException e) {
642 			throw new MojoExecutionException("Cannot read config file ["
643 					+ this.configFile + "]", e);
644 		} catch (SAXException e) {
645 			throw new MojoExecutionException("Cannot parse config file ["
646 					+ this.configFile + "]", e);
647 		} catch (ConfigReadException e) {
648 			throw new MojoExecutionException(e.getMessage(), e);
649 		} finally {
650 			if (configInput != null) {
651 				try {
652 					configInput.close();
653 				} catch (IOException e) {
654 					// ignore
655 				}
656 			}
657 		}
658 	}

660 	/**
661 	 * Returns the lineEnding parameter as characters when the value is known
662 	 * (LF, CRLF, CR) or can be determined from the file text (KEEP). Otherwise
663 	 * null is returned.
664 	 *
665 	 * @param fileDataString the file data string
666 	 * @return the line ending
667 	 */
668 	String getLineEnding(String fileDataString) {
669 		String lineEnd = null;
670 		if (LINE_ENDING_KEEP.equals(this.lineEnding)) {
671 			lineEnd = determineLineEnding(fileDataString);
672 		} else if (LINE_ENDING_LF.equals(this.lineEnding)) {
673 			lineEnd = LINE_ENDING_LF_CHAR;
674 		} else if (LINE_ENDING_CRLF.equals(this.lineEnding)) {
675 			lineEnd = LINE_ENDING_CRLF_CHARS;
676 		} else if (LINE_ENDING_CR.equals(this.lineEnding)) {
677 			lineEnd = LINE_ENDING_CR_CHAR;
678 		}
679 		return lineEnd;
680 	}

682 	/**
683 	 * Returns the most occurring line-ending characters in the file text or
684 	 * null if no line-ending occurs the most.
685 	 *
686 	 * @param fileDataString the file data string
687 	 * @return the string
688 	 */
689 	String determineLineEnding(String fileDataString) {
690 		int lfCount = 0;
691 		int crCount = 0;
692 		int crlfCount = 0;
693 
694 		for (int i = 0; i < fileDataString.length(); i++) {
695 			char c = fileDataString.charAt(i);
696 			if (c == '\r') {
697 				if ((i + 1) < fileDataString.length()
698 						&& fileDataString.charAt(i + 1) == '\n') {
699 					crlfCount++;
700 					i++;
701 				} else {
702 					crCount++;
703 				}
704 			} else if (c == '\n') {
705 				lfCount++;
706 			}
707 		}
708 		if (lfCount > crCount && lfCount > crlfCount) {
709 			return LINE_ENDING_LF_CHAR;
710 		} else if (crlfCount > lfCount && crlfCount > crCount) {
711 			return LINE_ENDING_CRLF_CHARS;
712 		} else if (crCount > lfCount && crCount > crlfCount) {
713 			return LINE_ENDING_CR_CHAR;
714 		}
715 		return null;
716 	}
