17  package net.revelc.code.formatter.javascript;
18  
19  import java.io.IOException;
20  import java.util.Map;
21  
22  import org.eclipse.jface.text.BadLocationException;
23  import org.eclipse.jface.text.Document;
24  import org.eclipse.jface.text.IDocument;
25  import org.eclipse.text.edits.TextEdit;
26  import org.eclipse.wst.jsdt.core.ToolFactory;
27  import org.eclipse.wst.jsdt.core.formatter.CodeFormatter;
28  
29  import net.revelc.code.formatter.AbstractCacheableFormatter;
30  import net.revelc.code.formatter.ConfigurationSource;
31  import net.revelc.code.formatter.Formatter;
32  import net.revelc.code.formatter.LineEnding;
33  
34  public class JavascriptFormatter extends AbstractCacheableFormatter implements Formatter {
35  
36      private CodeFormatter formatter;
37  
38      @Override
39      public void init(Map<String, String> options, ConfigurationSource cfg) {
40          super.initCfg(cfg);
41  
42          this.formatter = ToolFactory.createCodeFormatter(options);
43      }
44  
45      @Override
46      public String doFormat(String code, LineEnding ending) throws IOException, BadLocationException {
47          TextEdit te = this.formatter.format(CodeFormatter.K_JAVASCRIPT_UNIT, code, 0, code.length(), 0,
48                  ending.getChars());
49          if (te == null) {
50              this.log.debug(
51                      "Code cannot be formatted. Possible cause " + "is unmatched source/target/compliance version.");
52              return null;
53          }
54  
55          IDocument doc = new Document(code);
56          te.apply(doc);
57          String formattedCode = doc.get();
58  
59          if (code.equals(formattedCode)) {
60              return null;
61          }
62          return formattedCode;
63      }
64  
65      @Override
66      public boolean isInitialized() {
67          return formatter != null;
68      }
69  
70  }
