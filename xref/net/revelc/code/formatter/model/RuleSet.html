17  package net.revelc.code.formatter.model;
18  
19  import org.apache.commons.digester3.Digester;
20  import org.apache.commons.digester3.RuleSetBase;
21  
22  /**
23   * An Apache Commons Digester RuleSet for configuring a digester to parse the
24   * Eclipse formatter config XML into objects.
25   * 
26   * @author jecki
27   * @author Matt Blanchette
28   */
29  class RuleSet extends RuleSetBase {
30  
31      private static final String PROFILES_PROFILE = "profiles/profile";
32      private static final String PROFILES_PROFILE_SETTING = PROFILES_PROFILE + "/setting";
33  
34      /**
35       * Adds the rule instances.
36       *
37       * @param digester the digester
38       * @see org.apache.commons.digester3.RuleSetBase#addRuleInstances(org.apache.commons.digester3.Digester)
39       */
40      @Override
41      public void addRuleInstances(Digester digester) {
42          digester.addObjectCreate("profiles", Profiles.class);
43          digester.addObjectCreate(PROFILES_PROFILE, Profile.class);
44          digester.addObjectCreate(PROFILES_PROFILE_SETTING, Setting.class);
45  
46          digester.addSetNext(PROFILES_PROFILE, "addProfile");
47          digester.addSetNext(PROFILES_PROFILE_SETTING, "addSetting");
48  
49          digester.addSetProperties(PROFILES_PROFILE, "kind", "kind");
50          digester.addSetProperties(PROFILES_PROFILE_SETTING, "id", "id");
51          digester.addSetProperties(PROFILES_PROFILE_SETTING, "value", "value");
52      }
53  
54  }
