17  package net.revelc.code.formatter;
18  
19  import org.jsoup.Jsoup;
20  import org.jsoup.nodes.Document;
21  import org.jsoup.nodes.Document.OutputSettings.Syntax;
22  import org.jsoup.nodes.Entities.EscapeMode;
23  import org.jsoup.parser.Parser;
24  
25  import java.io.IOException;
26  import java.nio.charset.Charset;
27  import java.nio.charset.StandardCharsets;
28  import java.nio.file.Files;
29  import java.nio.file.Paths;
30  import java.util.Map;
31  import java.util.Properties;
32  
33  /**
34   * @author yoshiman
35   *
36   */
37  public abstract class JsoupBasedFormatter extends AbstractCacheableFormatter implements Formatter {
38  
39      private String filename;
40  
41      @Override
42      public void init(Map<String, String> options, ConfigurationSource cfg) {
43          if (cfg != null) {
44              super.initCfg(cfg);
45          }
46      }
47  
48      public String doFormat(String code, Syntax syntax) {
49          Document document;
50          switch (syntax) {
51          case html:
52              document = Jsoup.parse(code, "", Parser.htmlParser());
53              break;
54          case xml:
55              document = Jsoup.parse(code, "", Parser.xmlParser());
56              break;
57          default:
58              throw new IllegalArgumentException(syntax + " is not allowed as syntax");
59          }
60          try {
61              Properties properties = new Properties();
62              properties.load(Files.newInputStream(Paths.get(filename)));
63              Charset charset = Charset.forName(properties.getProperty("charset", StandardCharsets.UTF_8.name()));
64              EscapeMode escapeMode = EscapeMode.valueOf(properties.getProperty("escapeMode", EscapeMode.xhtml.name()));
65              int indentAmount = Integer.parseInt(properties.getProperty("indentAmount", "1"));
66              boolean outlineMode = Boolean.parseBoolean(properties.getProperty("outlineMode", Boolean.toString(true)));
67              boolean pretty = Boolean.parseBoolean(properties.getProperty("pretty", Boolean.toString(true)));
68              document.outputSettings() //
69                      .charset(charset) //
70                      .escapeMode(escapeMode) //
71                      .indentAmount(indentAmount) //
72                      .outline(outlineMode) //
73                      .prettyPrint(pretty) //
74                      .syntax(syntax);
75  
76          } catch (IOException e) {
77              this.log.error(e);
78          }
79  
80          String formattedCode = document.outerHtml();
81          if (code.equals(formattedCode)) {
82              return null;
83          }
84          return formattedCode;
85      }
86  
87      @Override
88      public boolean isInitialized() {
89          return filename != null;
90      }
91  
92      public String getFilename() {
93          return filename;
94      }
95  
96      public void setFilename(String filename) {
97          this.filename = filename;
98      }
99  
100 }
