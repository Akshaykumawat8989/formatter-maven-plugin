81      @Override
82      public void configure(ProjectConfigurationRequest request, IProgressMonitor monitor) throws CoreException {
83          IProject eclipseProject = request.getProject();
84  
85          printSettings();
86          if (eclipseProject.hasNature(JavaCore.NATURE_ID)) {
87              Xpp3Dom[] settings = parseConfigurationFile(request, monitor);
88  
89              for (Xpp3Dom setting : settings) {
90                  Platform.getPreferencesService().getRootNode().node("project").node(eclipseProject.getName())
91                          .node(JavaCore.PLUGIN_ID).put(setting.getAttribute("id"), setting.getAttribute("value"));
92              }
93  
94              Platform.getPreferencesService().getRootNode().node("project").node(eclipseProject.getName())
95                      .node("org.eclipse.jdt.ui").put("cleanup.format_source_code", "true");
96          }
97  
98          // jsdtConfigFile = cfg.getChild("configJsFile").getValue();
99          // src/config/eclipse/formatter/javascript.xml
100     }

102     private Xpp3Dom[] parseConfigurationFile(ProjectConfigurationRequest request, IProgressMonitor monitor)
103             throws CoreException {
104         Xpp3Dom dom;
105         try (InputStream content = readConfigFile(Formatter.JAVA, request, monitor)) {
106             dom = Xpp3DomBuilder.build(content, "UTF-8");
107         } catch (XmlPullParserException e) {
108             throw new CoreException(new Status(IStatus.ERROR, FormatterCore.PLUGIN_ID, "Invalid configuration XML", e));
109         } catch (IOException e) {
110             throw new CoreException(
111                     new Status(IStatus.ERROR, FormatterCore.PLUGIN_ID, "Unable to read configuration XML", e));
112         }
113         return dom.getChild("profile").getChildren("setting");
114     }
115 
116     private InputStream readConfigFile(Formatter formatter, ProjectConfigurationRequest request,
117             IProgressMonitor monitor) throws CoreException {
118         IMavenProjectFacade mavenProject = request.getMavenProjectFacade();
119         List<MojoExecution> executions = mavenProject.getMojoExecutions("net.revelc.code.formatter",
120                 "formatter-maven-plugin", monitor, "validate");
121 
122         MojoExecution execution = executions.get(0);
123         Xpp3Dom cfg = execution.getConfiguration();
124         String javaConfigFile = cfg.getChild(formatter.getConfigurationName()).getValue();
125         if (javaConfigFile == null || javaConfigFile.equalsIgnoreCase("${" + formatter.getConfigurationName() + "}")) {
126             javaConfigFile = formatter.getDefaultPath();
127         }
128 
129         File cfgFile = new File(javaConfigFile);
130         if (!cfgFile.isAbsolute()) {
131             cfgFile = request.getProject().getLocation().append(javaConfigFile).toFile();
132         }
133 
134         if (!cfgFile.exists())
135             throw new CoreException(new Status(IStatus.CANCEL, FormatterCore.PLUGIN_ID,
136                     "Configuration file '" + javaConfigFile + "' not found!"));
137 
138         try {
139             return new FileInputStream(cfgFile);
140         } catch (FileNotFoundException e) {
141             throw new CoreException(new Status(IStatus.CANCEL, FormatterCore.PLUGIN_ID, e.getMessage()));
142         }
143     }

145     private void printSettings() {
146         StringBuilder sb = new StringBuilder();
147         Preferences prefs = Platform.getPreferencesService().getRootNode();
148         try {
149             eval(prefs, "\t", sb);
150         } catch (Exception e1) {
151             LOGGER.info("Exception in eval " + e1.getMessage());
152         }
153 
154         File f = new File("tree.txt");
155         try (final OutputStreamWriter osw = new OutputStreamWriter(new FileOutputStream(f), Charset.forName("UTF-8"))) {
156             osw.write(sb.toString().toCharArray());
157             f.getAbsolutePath();
158         } catch (IOException e1) {
159             LOGGER.info("Exception in writing in tree.txt " + e1.getMessage());
160         }
161     }
162 
163     private static void eval(Preferences prefs, String spacer, StringBuilder sb) throws Exception {
164         String[] children = prefs.childrenNames();
165         for (String child : children) {
166             sb.append(spacer).append(child).append("\n");
167             eval(prefs.node(child), spacer + "\t", sb);
168         }
169         String[] keys = prefs.keys();
170         for (String key : keys) {
171             sb.append(spacer).append(" * ").append(key).append(": ").append(prefs.get(key, null)).append("\n");
172         }
173     }
