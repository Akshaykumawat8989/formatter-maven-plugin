17  package net.revelc.code.formatter.model;
18  
19  import java.io.IOException;
20  import java.io.InputStream;
21  import java.util.List;
22  import java.util.Map;
23  
24  import org.apache.commons.digester3.Digester;
25  import org.xml.sax.SAXException;
26  
27  /**
28   * This class reads a config file for Eclipse code formatter.
29   * 
30   * @author jecki
31   * @author Matt Blanchette
32   */
33  public class ConfigReader {
34  
35      /**
36       * Read from the <code>input</code> and return it's configuration settings
37       * as a {@link Map}.
38       *
39       * @param input the input
40       * @return return {@link Map} with all the configurations read from the
41       *         config file, or throws an exception if there's a problem reading
42       *         the input, e.g.: invalid XML.
43       * @throws IOException Signals that an I/O exception has occurred.
44       * @throws SAXException the SAX exception
45       * @throws ConfigReadException the config read exception
46       */
47      public Map<String, String> read(InputStream input) throws IOException, SAXException, ConfigReadException {
48          Digester digester = new Digester();
49          digester.addRuleSet(new RuleSet());
50  
51          Object result = digester.parse(input);
52          if (result == null || !(result instanceof Profiles)) {
53              throw new ConfigReadException("No profiles found in config file");
54          }
55  
56          Profiles profiles = (Profiles) result;
57          List<Map<String, String>> list = profiles.getProfiles();
58          if (list.isEmpty()) {
59              throw new ConfigReadException("No profile in config file of kind: " + Profiles.PROFILE_KIND);
60          }
61  
62          return list.get(0);
63      }
64  
65  }
